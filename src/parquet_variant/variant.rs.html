36const MAX_SHORT_STRING_BYTES: usize = 0x3F;
37
38/// A Variant [`ShortString`]
39///
40/// This implementation is a zero cost wrapper over `&str` that ensures
41/// the length of the underlying string is a valid Variant short string (63 bytes or less)
42#[derive(Debug, Clone, Copy, PartialEq)]
43pub struct ShortString<'a>(pub(crate) &'a str);
44
45impl<'a> ShortString<'a> {
46    /// Attempts to interpret `value` as a variant short string value.
47    ///
48    /// # Errors
49    ///
50    /// Returns an error if  `value` is longer than the maximum allowed length
51    /// of a Variant short string (63 bytes).
52    pub fn try_new(value: &'a str) -> Result<Self, ArrowError> {
53        if value.len() > MAX_SHORT_STRING_BYTES {
54            return Err(ArrowError::InvalidArgumentError(format!(
55                "value is larger than {MAX_SHORT_STRING_BYTES} bytes"
56            )));
57        }
58
59        Ok(Self(value))
60    }
61
62    /// Returns the underlying Variant short string as a &str
63    pub fn as_str(&self) -> &'a str {
64        self.0
65    }
66}
67
68impl<'a> From<ShortString<'a>> for &'a str {
69    fn from(value: ShortString<'a>) -> Self {
70        value.0
71    }
72}
73
74impl<'a> TryFrom<&'a str> for ShortString<'a> {
75    type Error = ArrowError;
76
77    fn try_from(value: &'a str) -> Result<Self, Self::Error> {
78        Self::try_new(value)
79    }
80}
81
82impl AsRef<str> for ShortString<'_> {
83    fn as_ref(&self) -> &str {
84        self.0
85    }
86}
87
88impl Deref for ShortString<'_> {
89    type Target = str;
90
91    fn deref(&self) -> &Self::Target {
92        self.0
93    }

96/// Represents a [Parquet Variant]
97///
98/// The lifetimes `'m` and `'v` are for metadata and value buffers, respectively.
99///
100/// # Background
101///
102/// The [specification] says:
103///
104/// The Variant Binary Encoding allows representation of semi-structured data
105/// (e.g. JSON) in a form that can be efficiently queried by path. The design is
106/// intended to allow efficient access to nested data even in the presence of
107/// very wide or deep structures.
108///
109/// Another motivation for the representation is that (aside from metadata) each
110/// nested Variant value is contiguous and self-contained. For example, in a
111/// Variant containing an Array of Variant values, the representation of an
112/// inner Variant value, when paired with the metadata of the full variant, is
113/// itself a valid Variant.
114///
115/// When stored in Parquet files, Variant fields can also be *shredded*. Shredding
116/// refers to extracting some elements of the variant into separate columns for
117/// more efficient extraction/filter pushdown. The [Variant Shredding
118/// specification] describes the details of shredding Variant values as typed
119/// Parquet columns.
120///
121/// A Variant represents a type that contains one of:
122///
123/// * Primitive: A type and corresponding value (e.g. INT, STRING)
124///
125/// * Array: An ordered list of Variant values
126///
127/// * Object: An unordered collection of string/Variant pairs (i.e. key/value
128///   pairs). An object may not contain duplicate keys.
129///
130/// # Encoding
131///
132/// A Variant is encoded with 2 binary values, the value and the metadata. The
133/// metadata stores a header and an optional dictionary of field names which are
134/// referred to by offset in the value. The value is a binary representation of
135/// the actual data, and varies depending on the type.
136///
137/// # Design Goals
138///
139/// The design goals of the Rust API are as follows:
140/// 1. Speed / Zero copy access (no `clone`ing is required)
141/// 2. Safety
142/// 3. Follow standard Rust conventions
143///
144/// [Parquet Variant]: https://github.com/apache/parquet-format/blob/master/VariantEncoding.md
145/// [specification]: https://github.com/apache/parquet-format/blob/master/VariantEncoding.md
146/// [Variant Shredding specification]: https://github.com/apache/parquet-format/blob/master/VariantShredding.md
147///
148/// # Examples:
149///
150/// ## Creating `Variant` from Rust Types
151/// ```
152/// use parquet_variant::Variant;
153/// // variants can be directly constructed
154/// let variant = Variant::Int32(123);
155/// // or constructed via `From` impls
156/// assert_eq!(variant, Variant::from(123i32));
157/// ```
158/// ## Creating `Variant` from metadata and value
159/// ```
160/// # use parquet_variant::{Variant, VariantMetadata};
161/// let metadata = [0x01, 0x00, 0x00];
162/// let value = [0x09, 0x48, 0x49];
163/// // parse the header metadata
164/// assert_eq!(
165///   Variant::from("HI"),
166///   Variant::new(&metadata, &value)
167/// );
168/// ```
169///
170/// ## Using `Variant` values
171/// ```
172/// # use parquet_variant::Variant;
173/// # let variant = Variant::Int32(123);
174/// // variants can be used in match statements like normal enums
175/// match variant {
176///   Variant::Int32(i) => println!("Integer: {}", i),
177///   Variant::String(s) => println!("String: {}", s),
178///   _ => println!("Other variant"),
179/// }
180/// ```
181///
182/// # Validation
183///
184/// Every instance of variant is either _valid_ or _invalid_. depending on whether the
185/// underlying bytes are a valid encoding of a variant value (see below).
186///
187/// Instances produced by [`Self::try_new`], [`Self::try_new_with_metadata`], or [`Self::validate`]
188/// are fully _validated_. They always contain _valid_ data, and infallible accesses such as
189/// iteration and indexing are panic-free. The validation cost is `O(m + v)` where `m` and
190/// `v` are the number of bytes in the metadata and value buffers, respectively.
191///
192/// Instances produced by [`Self::new`] and [`Self::new_with_metadata`] are _unvalidated_ and so
193/// they may contain either _valid_ or _invalid_ data. Infallible accesses to variant objects and
194/// arrays, such as iteration and indexing will panic if the underlying bytes are _invalid_, and
195/// fallible alternatives are provided as panic-free alternatives. [`Self::validate`] can also be
196/// used to _validate_ an _unvalidated_ instance, if desired.
197///
198/// _Unvalidated_ instances can be constructed in constant time. This can be useful if the caller
199/// knows the underlying bytes were already validated previously, or if the caller intends to
200/// perform a small number of (fallible) accesses to a large variant value.
201///
202/// A _validated_ variant value guarantees that the associated [metadata] and all nested [object]
203/// and [array] values are _valid_. Primitive variant subtypes are always _valid_ by construction.
204///
205/// # Safety
206///
207/// Even an _invalid_ variant value is still _safe_ to use in the Rust sense. Accessing it with
208/// infallible methods may cause panics but will never lead to undefined behavior.
209///
210/// [metadata]: VariantMetadata#Validation
211/// [object]: VariantObject#Validation
212/// [array]: VariantList#Validation
213#[derive(Debug, Clone, PartialEq)]
214pub enum Variant<'m, 'v> {
215    /// Primitive type: Null
216    Null,
217    /// Primitive (type_id=1): INT(8, SIGNED)
218    Int8(i8),
219    /// Primitive (type_id=1): INT(16, SIGNED)
220    Int16(i16),
221    /// Primitive (type_id=1): INT(32, SIGNED)
222    Int32(i32),
223    /// Primitive (type_id=1): INT(64, SIGNED)
224    Int64(i64),
225    /// Primitive (type_id=1): DATE
226    Date(NaiveDate),
227    /// Primitive (type_id=1): TIMESTAMP(isAdjustedToUTC=true, MICROS)
228    TimestampMicros(DateTime<Utc>),
229    /// Primitive (type_id=1): TIMESTAMP(isAdjustedToUTC=false, MICROS)
230    TimestampNtzMicros(NaiveDateTime),
231    /// Primitive (type_id=1): DECIMAL(precision, scale) 32-bits
232    Decimal4(VariantDecimal4),
233    /// Primitive (type_id=1): DECIMAL(precision, scale) 64-bits
234    Decimal8(VariantDecimal8),
235    /// Primitive (type_id=1): DECIMAL(precision, scale) 128-bits
236    Decimal16(VariantDecimal16),
237    /// Primitive (type_id=1): FLOAT
238    Float(f32),
239    /// Primitive (type_id=1): DOUBLE
240    Double(f64),
241    /// Primitive (type_id=1): BOOLEAN (true)
242    BooleanTrue,
243    /// Primitive (type_id=1): BOOLEAN (false)
244    BooleanFalse,
245    // Note: only need the *value* buffer for these types
246    /// Primitive (type_id=1): BINARY
247    Binary(&'v [u8]),
248    /// Primitive (type_id=1): STRING
249    String(&'v str),
250    /// Short String (type_id=2): STRING
251    ShortString(ShortString<'v>),
252    // need both metadata & value
253    /// Object (type_id=3): N/A
254    Object(VariantObject<'m, 'v>),
255    /// Array (type_id=4): N/A
256    List(VariantList<'m, 'v>),
257}

259impl<'m, 'v> Variant<'m, 'v> {
260    /// Attempts to interpret a metadata and value buffer pair as a new `Variant`.
261    ///
262    /// The instance is fully [validated].
263    ///
264    /// # Example
265    /// ```
266    /// use parquet_variant::{Variant, VariantMetadata};
267    /// let metadata = [0x01, 0x00, 0x00];
268    /// let value = [0x09, 0x48, 0x49];
269    /// // parse the header metadata
270    /// assert_eq!(
271    ///   Variant::from("HI"),
272    ///   Variant::try_new(&metadata, &value).unwrap()
273    /// );
274    /// ```
275    ///
276    /// [validated]: Self#Validation
277    pub fn try_new(metadata: &'m [u8], value: &'v [u8]) -> Result<Self, ArrowError> {
278        let metadata = VariantMetadata::try_new(metadata)?;
279        Self::try_new_with_metadata(metadata, value)
280    }
281
282    /// Attempts to interpret a metadata and value buffer pair as a new `Variant`.
283    ///
284    /// The instance is [unvalidated].
285    ///
286    /// # Example
287    /// ```
288    /// use parquet_variant::{Variant, VariantMetadata};
289    /// let metadata = [0x01, 0x00, 0x00];
290    /// let value = [0x09, 0x48, 0x49];
291    /// // parse the header metadata
292    /// assert_eq!(
293    ///   Variant::from("HI"),
294    ///   Variant::new(&metadata, &value)
295    /// );
296    /// ```
297    ///
298    /// [unvalidated]: Self#Validation
299    pub fn new(metadata: &'m [u8], value: &'v [u8]) -> Self {
300        let metadata = VariantMetadata::try_new_impl(metadata).expect("Invalid variant metadata");
301        Self::try_new_with_metadata_impl(metadata, value).expect("Invalid variant data")
302    }
303
304    /// Create a new variant with existing metadata.
305    ///
306    /// The instance is fully [validated].
307    ///
308    /// # Example
309    /// ```
310    /// # use parquet_variant::{Variant, VariantMetadata};
311    /// let metadata = [0x01, 0x00, 0x00];
312    /// let value = [0x09, 0x48, 0x49];
313    /// // parse the header metadata first
314    /// let metadata = VariantMetadata::new(&metadata);
315    /// assert_eq!(
316    ///   Variant::from("HI"),
317    ///   Variant::try_new_with_metadata(metadata, &value).unwrap()
318    /// );
319    /// ```
320    ///
321    /// [validated]: Self#Validation
322    pub fn try_new_with_metadata(
323        metadata: VariantMetadata<'m>,
324        value: &'v [u8],
325    ) -> Result<Self, ArrowError> {
326        Self::try_new_with_metadata_impl(metadata, value)?.validate()
327    }
328
329    /// Similar to [`Self::try_new_with_metadata`], but [unvalidated].
330    ///
331    /// [unvalidated]: Self#Validation
332    pub fn new_with_metadata(metadata: VariantMetadata<'m>, value: &'v [u8]) -> Self {
333        Self::try_new_with_metadata_impl(metadata, value).expect("Invalid variant")
334    }

336    // The actual constructor, which only performs shallow (constant-time) validation.
337    fn try_new_with_metadata_impl(
338        metadata: VariantMetadata<'m>,
339        value: &'v [u8],
340    ) -> Result<Self, ArrowError> {
341        let value_metadata = first_byte_from_slice(value)?;
342        let value_data = slice_from_slice(value, 1..)?;
343        let new_self = match get_basic_type(value_metadata) {
344            VariantBasicType::Primitive => match get_primitive_type(value_metadata)? {
345                VariantPrimitiveType::Null => Variant::Null,
346                VariantPrimitiveType::Int8 => Variant::Int8(decoder::decode_int8(value_data)?),
347                VariantPrimitiveType::Int16 => Variant::Int16(decoder::decode_int16(value_data)?),
348                VariantPrimitiveType::Int32 => Variant::Int32(decoder::decode_int32(value_data)?),
349                VariantPrimitiveType::Int64 => Variant::Int64(decoder::decode_int64(value_data)?),
350                VariantPrimitiveType::Decimal4 => {
351                    let (integer, scale) = decoder::decode_decimal4(value_data)?;
352                    Variant::Decimal4(VariantDecimal4::try_new(integer, scale)?)
353                }
354                VariantPrimitiveType::Decimal8 => {
355                    let (integer, scale) = decoder::decode_decimal8(value_data)?;
356                    Variant::Decimal8(VariantDecimal8::try_new(integer, scale)?)
357                }
358                VariantPrimitiveType::Decimal16 => {
359                    let (integer, scale) = decoder::decode_decimal16(value_data)?;
360                    Variant::Decimal16(VariantDecimal16::try_new(integer, scale)?)
361                }
362                VariantPrimitiveType::Float => Variant::Float(decoder::decode_float(value_data)?),
363                VariantPrimitiveType::Double => {
364                    Variant::Double(decoder::decode_double(value_data)?)
365                }
366                VariantPrimitiveType::BooleanTrue => Variant::BooleanTrue,
367                VariantPrimitiveType::BooleanFalse => Variant::BooleanFalse,
368                VariantPrimitiveType::Date => Variant::Date(decoder::decode_date(value_data)?),
369                VariantPrimitiveType::TimestampMicros => {
370                    Variant::TimestampMicros(decoder::decode_timestamp_micros(value_data)?)
371                }
372                VariantPrimitiveType::TimestampNtzMicros => {
373                    Variant::TimestampNtzMicros(decoder::decode_timestampntz_micros(value_data)?)
374                }
375                VariantPrimitiveType::Binary => {
376                    Variant::Binary(decoder::decode_binary(value_data)?)
377                }
378                VariantPrimitiveType::String => {
379                    Variant::String(decoder::decode_long_string(value_data)?)
380                }
381            },
382            VariantBasicType::ShortString => {
383                Variant::ShortString(decoder::decode_short_string(value_metadata, value_data)?)
384            }
385            VariantBasicType::Object => {
386                Variant::Object(VariantObject::try_new_impl(metadata, value)?)
387            }
388            VariantBasicType::Array => Variant::List(VariantList::try_new_impl(metadata, value)?),
389        };
390        Ok(new_self)
391    }

393    /// True if this variant instance has already been [validated].
394    ///
395    /// [validated]: Self#Validation
396    pub fn is_validated(&self) -> bool {
397        match self {
398            Variant::List(list) => list.is_validated(),
399            Variant::Object(obj) => obj.is_validated(),
400            _ => true,
401        }
402    }

404    /// Recursively validates this variant value, ensuring that infallible access will not panic due
405    /// to invalid bytes.
406    ///
407    /// Variant leaf values are always valid by construction, but [objects] and [arrays] can be
408    /// constructed in unvalidated (and potentially invalid) state.
409    ///
410    /// If [`Self::is_validated`] is true, validation is a no-op. Otherwise, the cost is `O(m + v)`
411    /// where `m` and `v` are the sizes of metadata and value buffers, respectively.
412    ///
413    /// [objects]: VariantObject#Validation
414    /// [arrays]: VariantList#Validation
415    pub fn validate(self) -> Result<Self, ArrowError> {
416        use Variant::*;
417        match self {
418            List(list) => list.validate().map(List),
419            Object(obj) => obj.validate().map(Object),
420            _ => Ok(self),
421        }
422    }

424    /// Converts this variant to `()` if it is null.
425    ///
426    /// Returns `Some(())` for null variants,
427    /// `None` for non-null variants.
428    ///
429    /// # Examples
430    ///
431    /// ```
432    /// use parquet_variant::Variant;
433    ///
434    /// // you can extract `()` from a null variant
435    /// let v1 = Variant::from(());
436    /// assert_eq!(v1.as_null(), Some(()));
437    ///
438    /// // but not from other variants
439    /// let v2 = Variant::from("hello!");
440    /// assert_eq!(v2.as_null(), None);
441    /// ```
442    pub fn as_null(&self) -> Option<()> {
443        matches!(self, Variant::Null).then_some(())
444    }

504    /// Converts this variant to a `DateTime<Utc>` if possible.
505    ///
506    /// Returns `Some(DateTime<Utc>)` for timestamp variants,
507    /// `None` for non-timestamp variants.
508    ///
509    /// # Examples
510    ///
511    /// ```
512    /// use parquet_variant::Variant;
513    /// use chrono::NaiveDate;
514    ///
515    /// // you can extract a DateTime<Utc> from a UTC-adjusted variant
516    /// let datetime = NaiveDate::from_ymd_opt(2025, 4, 16).unwrap().and_hms_milli_opt(12, 34, 56, 780).unwrap().and_utc();
517    /// let v1 = Variant::from(datetime);
518    /// assert_eq!(v1.as_datetime_utc(), Some(datetime));
519    ///
520    /// // or a non-UTC-adjusted variant
521    /// let datetime = NaiveDate::from_ymd_opt(2025, 4, 16).unwrap().and_hms_milli_opt(12, 34, 56, 780).unwrap();
522    /// let v2 = Variant::from(datetime);
523    /// assert_eq!(v2.as_datetime_utc(), Some(datetime.and_utc()));
524    ///
525    /// // but not from other variants
526    /// let v3 = Variant::from("hello!");
527    /// assert_eq!(v3.as_datetime_utc(), None);
528    /// ```
529    pub fn as_datetime_utc(&self) -> Option<DateTime<Utc>> {
530        match *self {
531            Variant::TimestampMicros(d) => Some(d),
532            Variant::TimestampNtzMicros(d) => Some(d.and_utc()),
533            _ => None,
534        }
535    }
